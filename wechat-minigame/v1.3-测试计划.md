## 测试用例

### 测试用例 1：顾客走动流畅性

**测试步骤：**
1. 启动小程序，停留在首页
2. 等待 2-4 秒，观察是否有顾客出现
3. 观察顾客走动是否流畅（不颤动）
4. 观察顾客是否绕开家具（如果有家具）
5. 观察顾客是否完成完整流程：进店→找座位→吃饭→付款→离开

**预期结果：**
- ✅ 顾客在 2-4 秒内出现
- ✅ 顾客移动流畅，速度适中（不颤动）
- ✅ 顾客遇到家具时绕行
- ✅ 顾客完成完整流程后离开

**验收标准：**
- 顾客移动速度 ≥ 3 像素/帧
- 更新频率 ≥ 20fps（50ms 间隔）
- 无明显卡顿或颤抖

## 自动化测试脚本

### 脚本 1：顾客走动测试

```javascript
// 在微信开发者工具控制台中运行

console.log('=== 顾客走动测试开始 ===');

// 1. 检查顾客生成
setTimeout(() => {
  const customers = GameGlobal.customers || [];
  console.log('顾客数量:', customers.length);
  
  if (customers.length > 0) {
    console.log('✅ 顾客生成正常');
    
    // 2. 检查顾客移动
    const customer = customers[0];
    const oldX = customer.x;
    const oldY = customer.y;
    
    setTimeout(() => {
      const newX = customer.x;
      const newY = customer.y;
      const dx = newX - oldX;
      const dy = newY - oldY;
      const dist = Math.sqrt(dx*dx + dy*dy);
      
      console.log('顾客移动距离:', dist.toFixed(2), '像素/50ms');
      
      if (dist > 0.5) {
        console.log('✅ 顾客移动流畅');
      } else {
        console.log('❌ 顾客移动异常（可能颤动）');
      }
    }, 50);
  } else {
    console.log('❌ 顾客生成异常');
  }
}, 2000);

console.log('=== 测试进行中... ===');
```

### 脚本 2：子导航点击测试

```javascript
// 在微信开发者工具控制台中运行

console.log('=== 子导航点击测试开始 ===');

// 模拟点击事件
function simulateTouch(x, y) {
  console.log(`模拟点击：(${x.toFixed(1)}, ${y.toFixed(1)})`);
  
  const action = CanvasRenderer.handleTouch(x, y, GameGlobal, 'business');
  console.log('返回动作:', action);
  
  return action;
}

// 测试经营 Tab 子导航（4 个按钮）
const subNavY = 84;  // 渲染位置
const subTabWidth = (375 - 32) / 4;

const testPoints = [
  { x: 16 + subTabWidth * 0.5, y: subNavY + 18, name: '餐厅' },
  { x: 16 + subTabWidth * 1.5, y: subNavY + 18, name: '厨房' },
  { x: 16 + subTabWidth * 2.5, y: subNavY + 18, name: '采购' },
  { x: 16 + subTabWidth * 3.5, y: subNavY + 18, name: '外卖' }
];

let successCount = 0;

testPoints.forEach((point, index) => {
  setTimeout(() => {
    const action = simulateTouch(point.x, point.y);
    
    if (action && action.type === 'subtab') {
      console.log(`✅ ${point.name} 点击成功`);
      successCount++;
    } else {
      console.log(`❌ ${point.name} 点击失败`);
    }
    
    if (index === testPoints.length - 1) {
      console.log(`\n测试结果：${successCount}/${testPoints.length} 成功`);
    }
  }, index * 500);
});

console.log('=== 测试进行中... ===');
```

const empTabs = [
  { key: 'chefs', label: '👨‍🍳 厨师' },
  { key: 'waiters', label: '👩‍🍳 服务员' },
  { key: 'cashiers', label: '🧾 收银员' }
];

empTabs.forEach((tab, index) => {
  const x = 16 + index * empTabWidth;

});

// 测试点击
function testEmpTab(index) {
  const x = 16 + index * empTabWidth + empTabWidth / 2;
  const y = empTabBaseY + 16;
  
  console.log(`\n测试点击：${empTabs[index].label}`);
  console.log(`点击坐标：(${x.toFixed(1)}, ${y.toFixed(1)})`);
  
  const action = CanvasRenderer.handleTouch(x, y, GameGlobal, 'business');
  console.log('返回动作:', action);
  
  if (action && action.action === 'selectEmpTab') {
    console.log(`✅ ${empTabs[index].label} 点击成功`);
    GameGlobal.selectedEmpTab = action.empTab;
  } else {
    console.log(`❌ ${empTabs[index].label} 点击失败`);
  }
}

// 依次测试三个 Tab
setTimeout(() => testEmpTab(0), 0);
setTimeout(() => testEmpTab(1), 500);
setTimeout(() => testEmpTab(2), 1000);
